# 头像

## 类型
支持三种类型：图片、Icon 以及字符，其中 Icon 和字符型可以自定义图标颜色及背景色。

<ShowSouceCode
  element={
    <Flex x='start'>
      <Avatar>U</Avatar>
      <Avatar icon={<MiIcon.User/>}></Avatar>
      <Avatar src={img} round='round'></Avatar>
      <Avatar style={{ backgroundColor: '#fde3cf', color: '#f56a00' }}>U</Avatar>
      <Avatar style={{ backgroundColor: '#87d068' }} icon={<MiIcon.User style={{color:'yellow'}}/>} />
    </Flex>
  }
  code={`
    import { Avatar, MiIcon , Flex } from 'miniEx';
    import React from 'react';
    import img from '../assets/logo.png'

    const App: React.FC = () => (
    <Flex x='start'>
      <Avatar>U</Avatar>
      <Avatar icon={<MiIcon.User/>}></Avatar>
      <Avatar src={img} round='round'></Avatar>
      <Avatar style={{ backgroundColor: '#fde3cf', color: '#f56a00' }}>U</Avatar>
      <Avatar style={{ backgroundColor: '#87d068' }} icon={<MiIcon.User style={{color:'yellow'}}/>} />
    </Flex>
    );

    export default App;
    `
  }
  language="ts"
  />

## 头像组
头像组合展现。

<ShowSouceCode
  element={AvatarGroupCN()}
  code={`
    import { AvatarGroup, MiIcon , Flex } from 'miniEx';
    import React from 'react';

    const App: React.FC = () => (
      const options = [
        {
          label: 'U',
          value: '1',
          style: { backgroundColor: '#fde3cf', color: '#f56a00' },
        },
        {
          icon: <MiIcon.User />,
          value: '2',
          style: { backgroundColor: '#87d068' },
        },
        {
          src: 'https://q1.qlogo.cn/g?b=qq&s=100&nk=1057072668',
          value: '3',
          round: 'round',
        },
        {
          label: 'k',
          value: '4',
        },
      ];
    return (
    <Flex direction='y'>
      <AvatarGroup options={options}></AvatarGroup>
      <AvatarGroup maxCount={2} options={options}></AvatarGroup>
    </Flex>
  );

    export default App;
    `
  }
  language="ts"
  />

# Button

## Basic

<Button btnType='default'>default</Button>
<Button btnType='primary'>primary</Button>
<Button btnType='danger'>danger</Button>
<Button btnType='link'>link</Button>

```ts
import { Button } from 'miniEx';
import React from 'react';

const App: React.FC = () => (
  <div>
    <Button btnType='default'>default</Button>
    <Button btnType='primary'>primary</Button>
    <Button btnType='danger'>danger</Button>
    <Button btnType='link'>link</Button>
  </div>
);

export default App;
```

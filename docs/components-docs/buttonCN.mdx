# 按钮

## 基础

<Button btnType='default'>default按钮</Button>
<Button btnType='primary'>primary按钮</Button>
<Button btnType='danger'>danger按钮</Button>
<Button btnType='link'>link按钮</Button>

```ts
import { Button } from 'miniEx';
import React from 'react';

const App: React.FC = () => (
  <div>
    <Button btnType='default'>default按钮</Button>
    <Button btnType='primary'>primary按钮</Button>
    <Button btnType='danger'>danger按钮</Button>
    <Button btnType='link'>link按钮</Button>
  </div>
);

export default App;
```
